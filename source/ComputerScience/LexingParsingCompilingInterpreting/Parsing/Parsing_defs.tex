\begin{smodule}[title={Parsing}]{Parsing}

    \symdecl*{Parser}
    \begin{sdefinition}[for=Parser]
        \definame{Parser}: A tool that interprets a sequence of tokens generated by the lexer and converts it into a parse tree based on a formal grammar.
    \end{sdefinition}

    \symdecl*{ParseTree}
    \begin{sdefinition}[for=ParseTree]
        \definame{ParseTree}: A tree representation of the syntactic structure of a source code, based on the rules of a formal grammar.
    \end{sdefinition}

    \symdecl*{LLParser}
    \begin{sdefinition}[for=LLParser]
        \definame{LLParser}: A top-down parser that reads input from left to right and produces a leftmost derivation.
    \end{sdefinition}

    \symdecl*{LRParser}
    \begin{sdefinition}[for=LRParser]
        \definame{LRParser}: A bottom-up parser that reads input from left to right and produces a rightmost derivation in reverse.
    \end{sdefinition}

    \symdecl*{Ambiguity}
    \begin{sdefinition}[for=Ambiguity]
        \definame{Ambiguity}: A situation where a grammar allows multiple valid parse trees for the same input.
    \end{sdefinition}

\end{smodule}
